#include <stdio.h>
#include <stdlib.h>

#define MAX 100000

// two stacks
int stack1[MAX], stack2[MAX];
int top1 = -1, top2 = -1;

// stack operations
void push1(int x) {
    stack1[++top1] = x;
}

int pop1() {
    return stack1[top1--];
}

void push2(int x) {
    stack2[++top2] = x;
}

int pop2() {
    return stack2[top2--];
}

// queue operations using two stacks
void enqueue(int x) {
    push1(x);
}

void shiftStacks() {
    if (top2 == -1) {  
        // move elements from stack1 to stack2
        while (top1 != -1) {
            push2(pop1());
        }
    }
}

void dequeue() {
    shiftStacks();
    if (top2 != -1) pop2();
}

void printFront() {
    shiftStacks();
    if (top2 != -1) {
        printf("%d\n", stack2[top2]);
    }
}

int main() {
    int q;
    scanf("%d", &q);

    while (q--) {
        int type, x;
        scanf("%d", &type);

        if (type == 1) {
            scanf("%d", &x);
            enqueue(x);
        } else if (type == 2) {
            dequeue();
        } else if (type == 3) {
            printFront();
        }
    }

    return 0;
}
