#include <stdio.h>
#include <stdlib.h>
#include <string.h>

char* isValid(char* s) {
    static char yes[] = "YES";
    static char no[] = "NO";
    
    int freq[26] = {0};
    for (int i = 0; s[i]; i++) {
        freq[s[i] - 'a']++;
    }

    int counts[26], n = 0;
    for (int i = 0; i < 26; i++) {
        if (freq[i] > 0) counts[n++] = freq[i];
    }

    // Find min and max frequency
    int min = counts[0], max = counts[0];
    for (int i = 1; i < n; i++) {
        if (counts[i] < min) min = counts[i];
        if (counts[i] > max) max = counts[i];
    }

    if (min == max) return yes;

    int minCount = 0, maxCount = 0;
    for (int i = 0; i < n; i++) {
        if (counts[i] == min) minCount++;
        if (counts[i] == max) maxCount++;
    }

    // Case 1: one char has frequency 1, remove it
    if (min == 1 && minCount == 1 && maxCount == n - 1) return yes;

    // Case 2: one char has frequency max, and max = min + 1
    if (max == min + 1 && maxCount == 1) return yes;

    return no;
}

int main() {
    char s[100005];
    scanf("%s", s);
    printf("%s\n", isValid(s));
    return 0;
}
