#include <stdio.h>
#include <stdlib.h>

int compare(const void *a, const void *b) {
    return (*(int *)a - *(int *)b);
}

int hackerlandRadioTransmitters(int x_count, int* x, int k) {
    qsort(x, x_count, sizeof(int), compare); // Sort house positions
    int count = 0;
    int i = 0;

    while (i < x_count) {
        count++;

        // Step 1: Move to the farthest house within k from current
        int start = x[i];
        while (i < x_count && x[i] <= start + k) {
            i++;
        }

        // Step 2: Place transmitter at last valid house
        int pos = x[i - 1];

        // Step 3: Skip houses within k from the transmitter
        while (i < x_count && x[i] <= pos + k) {
            i++;
        }
    }

    return count;
}

int main() {
    int n, k;
    scanf("%d %d", &n, &k);

    int *x = (int *)malloc(n * sizeof(int));
    for (int i = 0; i < n; i++) {
        scanf("%d", &x[i]);
    }

    int result = hackerlandRadioTransmitters(n, x, k);
    printf("%d\n", result);

    free(x);
    return 0;
}
