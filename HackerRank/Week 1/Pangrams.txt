#include <stdio.h>
#include <string.h>
#include <ctype.h>
char* pangrams(char* s) {
    int letters[26] = {0};
    int i;
    for (i = 0; s[i] != '\0'; i++) {
        if (isalpha(s[i])) {
            int index = tolower(s[i]) - 'a';
            letters[index] = 1;
        }
    }
    for (i = 0; i < 26; i++) {
        if (letters[i] == 0) {
            return "not pangram";
        }
    }
    return "pangram";
}
int main() {
    char s[1000];
    fgets(s, sizeof(s), stdin);
    size_t len = strlen(s);
    if (len > 0 && s[len - 1] == '\n') {
        s[len - 1] = '\0';
    }
    printf("%s\n", pangrams(s));
    return 0;
}
