#include <stdio.h>
#include <stdlib.h>

int cAsc(const void* a, const void* b) {
    return (*(int*)a - *(int*)b);
}
int cDesc(const void* a, const void* b) {
    return (*(int*)b - *(int*)a);
}
char* Arrays(int k, int A[], int B[], int n) {
    qsort(A, n, sizeof(int), cAsc);
    qsort(B, n, sizeof(int), cDesc);

    for (int i = 0; i < n; i++) {
        if (A[i] + B[i] < k) {
            return "NO";
        }
    }
    return "YES";
}
int main() {
    int q;
    scanf("%d", &q);
    while (q--) {
        int n, k;
        scanf("%d %d", &n, &k);
        int A[n], B[n];
        for (int i = 0; i < n; i++) {
            scanf("%d", &A[i]);
        }
        for (int i = 0; i < n; i++) {
            scanf("%d", &B[i]);
        }
        printf("%s\n", Arrays(k, A, B, n));
    }
    return 0;
}
