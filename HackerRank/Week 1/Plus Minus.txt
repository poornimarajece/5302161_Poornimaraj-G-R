#include <stdio.h>
void plusMinus(int a_c,int* a) {
    int positive = 0, negative = 0, zero = 0;
    for (int i = 0; i < a_c; i++) {
        if (a[i] > 0)
            positive++;
        else if (a[i] < 0)
            negative++;
        else
            zero++;
    }
    printf("%.6f\n", (float)positive / a_c);
    printf("%.6f\n", (float)negative / a_c);
    printf("%.6f\n", (float)zero / a_c);
}
int main() {
    int n;
    scanf("%d", &n);
    int a[n];
    for (int i = 0; i < n; i++) {
        scanf("%d", &a[i]);
    }
    plusMinus(n, a);
    return 0;
}
