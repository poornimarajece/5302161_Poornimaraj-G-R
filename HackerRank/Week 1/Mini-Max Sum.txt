#include <stdio.h>
void miniMaxSum(int a[5]) {
    long long total = 0;
    int min = a[0];
    int max = a[0];
    for (int i = 0; i < 5; i++) {
        total += a[i];
        if (a[i] < min) {
            min = a[i];
        }
        if (a[i] > max) {
            max = a[i];
        }
    }
    long long min_sum = total - max;
    long long max_sum = total - min;

    printf("%lld %lld\n", min_sum, max_sum);
}
int main() {
    int a[5];
    for (int i = 0; i < 5; i++) {
        scanf("%d", &a[i]);
    }
    miniMaxSum(a);
    return 0;
}
