#include <stdio.h>
#include <string.h>
#include <stdlib.h>

char* timeConversion(char* s) {
    static char result[9];
    int h;
    char p[3];
    sscanf(s, "%2d:%*2d:%*2d%2s", &h, p);

    if (strncmp(&s[8], "AM", 2) == 0) {
        if (h == 12) h = 0;
    } else if (strncmp(&s[8], "PM", 2) == 0) {
        if (h != 12) h += 12;
    }
    snprintf(result, sizeof(result), "%02d:%c%c:%c%c", 
             h, s[3], s[4], s[6], s[7]);

    return result;
}
int main() {
    char s[11];
    scanf("%s", s);
    char* result = timeConversion(s);
    printf("%s\n", result);
    return 0;
}
