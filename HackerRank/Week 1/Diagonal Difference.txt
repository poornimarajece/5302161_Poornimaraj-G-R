#include <stdio.h>
#include <stdlib.h>
int diagonalDifference(int n, int arr[n][n]) {
    int primarySum = 0;
    int secondarySum = 0;
    for (int i = 0; i < n; i++) {
        primarySum += arr[i][i];               
        secondarySum += arr[i][n - i - 1];     
    }
    return abs(primarySum - secondarySum);
}
int main() {
    int n;
    scanf("%d", &n);
    int a[n][n];
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            scanf("%d", &a[i][j]);
        }
    }
    int result = diagonalDifference(n, a);
    printf("%d\n", result);
    return 0;
}
